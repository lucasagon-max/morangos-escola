async function carregarPedidos() { try { const { data, error } = await supabaseClient .from('pedidos_escola') .select('*') .order('created_at', { ascending: false }); if (error) throw error; let html = ''; let totalArrecadado = 0; let totalBandejasPagas = 0; let pedidosPendentes = 0; if (data.length === 0) { document.getElementById('corpoTabela').innerHTML = 'Nenhum pedido feito ainda.'; return; } data.forEach(pedido => { const dataObj = new Date(pedido.created_at); const dataFormatada = dataObj.toLocaleDateString('pt-BR') + ' ' + dataObj.toLocaleTimeString('pt-BR', {hour: '2-digit', minute:'2-digit'}); const isPago = pedido.status_pagamento === 'pago'; const statusClass = isPago ? 'bg-green-100 text-green-700' : 'bg-orange-100 text-orange-700'; const statusTexto = isPago ? 'PAGO' : 'PENDENTE'; const linhaClass = isPago ? '' : 'linha-pendente'; // === LÓGICA DO WHATSAPP === // Limpa o telefone (tira os parênteses e traços) para o link funcionar const telefoneLimpo = pedido.telefone_responsavel.replace(/\D/g, ''); // Texto automático de marketing const mensagemZap = `Olá, ${pedido.nome_responsavel}! 🍓 Passando para confirmar que recebemos certinho o pagamento da(s) bandeja(s) do(a) ${pedido.nome_aluno}. Muito obrigado! 🥰\n\nSe você achou o nosso sistema prático, ajuda a gente espalhando o link no grupo de pais da escola? É esse aqui: https://clientes.morangostatui.com.br 🚀`; // Cria o link do botão const linkZap = `https://wa.me/55${telefoneLimpo}?text=${encodeURIComponent(mensagemZap)}`; if (isPago) { totalArrecadado += pedido.valor_total; totalBandejasPagas += pedido.quantidade; } else { pedidosPendentes++; } html += ` ${dataFormatada} ${pedido.nome_aluno} ${pedido.turma}
${pedido.nome_responsavel}
${pedido.telefone_responsavel} ${isPago ? `💬 Avisar no Zap` : ''}
${pedido.quantidade} ${statusTexto}
`; }); document.getElementById('corpoTabela').innerHTML = html; document.getElementById('cardValor').innerText = 'R$ ' + totalArrecadado.toFixed(2).replace('.', ','); document.getElementById('cardBandejas').innerText = totalBandejasPagas; document.getElementById('cardPendentes').innerText = pedidosPendentes + ' pedidos'; } catch (error) { console.error(error); document.getElementById('corpoTabela').innerHTML = 'Erro de segurança ou ao carregar os dados.'; } }
